-- Strategic Crisis Simulation Platform
-- Database Schema (PostgreSQL / Supabase)

-- Users table
CREATE TABLE users (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    email VARCHAR(255) UNIQUE NOT NULL,
    created_at TIMESTAMP DEFAULT NOW(),
    updated_at TIMESTAMP DEFAULT NOW()
);

-- Sessions table
CREATE TABLE sessions (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    user_id UUID REFERENCES users(id) ON DELETE CASCADE,
    scenario_type VARCHAR(100) NOT NULL DEFAULT 'supply_chain',
    scenario_seed FLOAT NOT NULL,
    scenario_data JSONB NOT NULL,
    initial_state JSONB NOT NULL,
    final_state JSONB,
    final_score INTEGER,
    performance_tier VARCHAR(50),
    created_at TIMESTAMP DEFAULT NOW(),
    completed_at TIMESTAMP
);

-- Turns table (stores each decision in a session)
CREATE TABLE turns (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    session_id UUID REFERENCES sessions(id) ON DELETE CASCADE,
    turn_number INTEGER NOT NULL,
    state_snapshot JSONB NOT NULL,
    decision_action VARCHAR(100) NOT NULL,
    consequences JSONB NOT NULL,
    event_triggered VARCHAR(100),
    created_at TIMESTAMP DEFAULT NOW()
);

-- User profiles (behavioral analysis)
CREATE TABLE user_profiles (
    user_id UUID PRIMARY KEY REFERENCES users(id) ON DELETE CASCADE,
    total_simulations INTEGER DEFAULT 0,
    average_score FLOAT,
    risk_tolerance_score FLOAT,
    decision_speed_score FLOAT,
    bias_patterns JSONB DEFAULT '[]'::jsonb,
    strengths JSONB DEFAULT '[]'::jsonb,
    weaknesses JSONB DEFAULT '[]'::jsonb,
    updated_at TIMESTAMP DEFAULT NOW()
);

-- Feedback reports
CREATE TABLE feedback_reports (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    session_id UUID REFERENCES sessions(id) ON DELETE CASCADE,
    final_score INTEGER NOT NULL,
    survival_probability FLOAT,
    performance_tier VARCHAR(50),
    decision_analysis JSONB NOT NULL,
    bias_analysis JSONB NOT NULL,
    optimal_strategy JSONB NOT NULL,
    counterfactual JSONB NOT NULL,
    lessons JSONB NOT NULL,
    created_at TIMESTAMP DEFAULT NOW()
);

-- Indexes for performance
CREATE INDEX idx_sessions_user_id ON sessions(user_id);
CREATE INDEX idx_sessions_completed_at ON sessions(completed_at);
CREATE INDEX idx_turns_session_id ON turns(session_id);
CREATE INDEX idx_feedback_reports_session_id ON feedback_reports(session_id);

-- Function to update user profile after each simulation
CREATE OR REPLACE FUNCTION update_user_profile()
RETURNS TRIGGER AS $$
BEGIN
    -- Update user profile statistics
    UPDATE user_profiles
    SET 
        total_simulations = total_simulations + 1,
        average_score = (
            SELECT AVG(final_score)
            FROM sessions
            WHERE user_id = NEW.user_id AND final_score IS NOT NULL
        ),
        updated_at = NOW()
    WHERE user_id = NEW.user_id;
    
    -- Insert profile if it doesn't exist
    INSERT INTO user_profiles (user_id, total_simulations)
    VALUES (NEW.user_id, 1)
    ON CONFLICT (user_id) DO NOTHING;
    
    RETURN NEW;
END;
$$ LANGUAGE plpgsql;

-- Trigger to automatically update user profile
CREATE TRIGGER trigger_update_user_profile
AFTER UPDATE OF final_score ON sessions
FOR EACH ROW
WHEN (NEW.final_score IS NOT NULL)
EXECUTE FUNCTION update_user_profile();
